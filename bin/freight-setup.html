freight-setup

Track down Freight's library path.

Add the library and types paths to PATH to allow sourceing library code.

#!/bin/sh

set -e

LIB="$(dirname $(dirname $0))/lib/freight"

source the configuration.

PATH="$LIB:$LIB/types:$PATH"


. conf.sh

log() {
	[ -n "$VERBOSE" ] && echo $* >&2
}
